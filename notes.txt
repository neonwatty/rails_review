# create app with postgres and tailwind
rails new MY_APP -d=postgresql --css=tailwind

# create action storage record tables
rails active_storage:install
rails db:migrate

# create user (for backend use)
### just use devise
# add it to gemfile
bundle add devise

# then follow setup instructions
...

# create User model via devise
rails generate devise User

# rails g model User email:string password_digest:string

# go to rails console, create user
User.create({email: "neonwatty@gmail.com", password: "MY_PASSWORD", password_confirmation: "MY_PASSWORD"})

# create upload with active attachment
rails g model Upload user:belongs_to video:attachment

# create Status table
rails g model Status upload:belongs_to delivery:string preprocess:string action:string

# create Output table
rails g model Output upload:belongs_to result:string

# running rails tests
rails db:create RAILS_ENV=test 
rails db:migrate RAILS_ENV=test
rails db:seed RAILS_ENV=test

or just 

rails db:reset RAILS_ENV=test

# when using devise
make sure to add devise helpers to test/test_helper.rb e.g., for integration
    include Devise::Test::IntegrationHelpers
